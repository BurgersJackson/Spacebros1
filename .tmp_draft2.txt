	                    ctx.setLineDash([]);

	                }

	                    ctx.lineWidth = 16 / z; 
	                    ctx.strokeStyle = 'rgba(0, 55, 100, 0.90)';

	        class CaveDraftZone extends Entity { 
	            constructor(x, y, w, h, forceY = -0.08) { 
	                super(x, y); 
	                this.w = w; 
	                this.h = h; 
	                this.forceY = forceY; 
	                this.t = 0; 
	            } 
	            contains(entity) { 
	                if (!entity || entity.dead) return false; 
	                return (entity.pos.x > this.pos.x - this.w/2 && entity.pos.x < this.pos.x + this.w/2 && entity.pos.y > this.pos.y - this.h/2 && entity.pos.y < this.pos.y + this.h/2); 
	            } 
	            update() { 
	                this.t++; 
	                const apply = (e) => { 
	                    if (!this.contains(e)) return; 
	                    e.vel.y += this.forceY; 
	                    // Slight stabilization so it feels like a current. 
	                    e.vel.x *= 0.995; 
	                }; 
	                apply(player); 
	                for (let i = 0; i < enemies.length; i++) apply(enemies[i]); 
	            } 
	            draw(ctx) { 
	                const z = currentZoom || ZOOM_LEVEL; 
	                const pulse = 0.35 + Math.abs(Math.sin(this.t * 0.04)) * 0.35; 
	                ctx.save(); 
	                ctx.translate(this.pos.x, this.pos.y); 
	                ctx.globalAlpha = 0.35; 
	                ctx.strokeStyle = `rgba(0,255,255,${0.08 + pulse * 0.08})`; 
	                ctx.lineWidth = 2 / z; 
	                ctx.strokeRect(-this.w/2, -this.h/2, this.w, this.h); 
	                ctx.globalAlpha = 0.18; 
	                ctx.fillStyle = '#0ff'; 
	                for (let i = -2; i <= 2; i++) { 
	                    const y = (i * 120) + Math.sin(this.t * 0.06 + i) * 30; 
	                    ctx.beginPath(); 
	                    ctx.moveTo(-this.w/2 + 20, y); 
	                    ctx.lineTo(this.w/2 - 20, y); 
	                    ctx.stroke(); 
	                } 
	                ctx.globalAlpha = 1; 
	                ctx.restore(); 
	            } 
	        } 
	         
	        class CaveCritter extends Entity { 
	            constructor(x, y) { 
	                super(x, y); 
	                this.radius = 14; 
	                this.t = 0; 
	                this.vel.x = (Math.random() - 0.5) * 2.2; 
	                this.vel.y = (Math.random() - 0.5) * 2.2; 
	            } 
	            scatter(fromX, fromY) { 
	                const a = Math.atan2(this.pos.y - fromY, this.pos.x - fromX); 
	                this.vel.x += Math.cos(a) * 4.2; 
	                this.vel.y += Math.sin(a) * 4.2; 
	            } 
	            update() { 
	                if (this.dead) return; 
	                this.t++; 
	                if (player && !player.dead) { 
	                    const d = Math.hypot(player.pos.x - this.pos.x, player.pos.y - this.pos.y); 
	                    if (d < 520) this.scatter(player.pos.x, player.pos.y); 
	                } 
	                this.vel.x *= 0.98; 
	                this.vel.y *= 0.98; 
	                this.pos.add(this.vel); 
	                checkWallCollision(this, 0.35); 
	            } 
	            draw(ctx) { 
	                if (this.dead) return; 
	                const z = currentZoom || ZOOM_LEVEL; 
	                ctx.save(); 
	                ctx.translate(this.pos.x, this.pos.y); 
	                ctx.shadowBlur = 10; 
	                ctx.shadowColor = '#6f6'; 
	                ctx.fillStyle = '#3c3'; 
	                ctx.strokeStyle = '#0a0'; 
	                ctx.lineWidth = 2 / z; 
	                ctx.beginPath(); 
	                ctx.arc(0, 0, this.radius, 0, Math.PI * 2); 
	                ctx.fill(); 
	                ctx.stroke(); 
	                ctx.shadowBlur = 0; 
	                ctx.restore(); 
	            } 
	        }
