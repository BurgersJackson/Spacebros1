            constructor(x, y, type = 'standard') {
                super(0, 0);
                this.pos.x = x;
                this.pos.y = y;
                this.type = type; 
                this.radius = 70; 
                this.hp = 10 + (difficultyTier - 1) * 5; 
                this.shootTimer = 150;
                this.angle = 0;
                this.turretAngle = 0;
                this.shieldRadius = 110;
                this.aggro = false; 
                
                let outerCount = 24;
                let outerHp = 1;
                let innerCount = 0;
                let innerHp = 0;

                if (difficultyTier === 1) { outerCount = 12; outerHp = 1; }
                else if (difficultyTier === 2) { outerCount = 16; outerHp = 1; }
                else if (difficultyTier === 3) { outerCount = 24; outerHp = 1; }
                else if (difficultyTier === 4) { outerCount = 24; outerHp = 2; innerCount = 8; innerHp = 1; }
                else if (difficultyTier === 5) { outerCount = 24; outerHp = 2; innerCount = 12; innerHp = 2; }
                else if (difficultyTier >= 6) { 
                    outerCount = 24; 
                    outerHp = 3 + (difficultyTier - 6); 
                    innerCount = 16 + (difficultyTier - 6); 
                    innerHp = 2 + Math.floor((difficultyTier - 6)/2); 
                }

                if (type === 'heavy') {
                    this.hp *= 1.5;
                    outerHp = Math.ceil(outerHp * 1.5);
                    innerHp = Math.ceil(innerHp * 1.5);
                    this.shootTimer = 120; 
                } else if (type === 'rapid') {
                    this.hp *= 0.7;
                    outerHp = Math.max(1, Math.floor(outerHp * 0.8));
                    this.shootTimer = 30; 
                }

                this.maxShieldHp = outerHp; 
                this.shieldSegments = new Array(outerCount).fill(outerHp); 
                this.shieldRotation = 0;
                
                this.innerShieldRadius = 90;
                this.innerShieldSegments = [];
                if (innerCount > 0) {
                    this.innerShieldSegments = new Array(innerCount).fill(innerHp);
                }
                this.innerShieldRotation = 0;
                
                const angle = Math.random() * Math.PI * 2;
                let speed = 0.2 + Math.random() * 0.3;
                if (type === 'heavy') speed *= 0.5;
                if (type === 'rapid') speed *= 1.5;
                
                this.vel.x = Math.cos(angle) * speed;
                this.vel.y = Math.sin(angle) * speed;
            }
            
            update() {
                if (player && !player.dead) {
                    const dx = player.pos.x - this.pos.x;
                    const dy = player.pos.y - this.pos.y;
                    const dist = Math.hypot(dx, dy);

                    // Keep distance from active cruiser boss
                    let dreadAvoidX = 0, dreadAvoidY = 0;
                    if (bossActive && boss && boss.isCruiser && !boss.dead) {
                        const bdx = boss.pos.x - this.pos.x;
                        const bdy = boss.pos.y - this.pos.y;
                        const bdist = Math.hypot(bdx, bdy);
                        if (bdist < 200) {
                            dreadAvoidX = -(bdx / (bdist || 1)) * 0.2;
                            dreadAvoidY = -(bdy / (bdist || 1)) * 0.2;
                        }
                    }

                    // Avoid bunching with other bases
                    for (let b of bases) {
                        if (b === this || b.dead) continue;
                        const bx = b.pos.x - this.pos.x;
                        const by = b.pos.y - this.pos.y;
                        const bdist = Math.hypot(bx, by);
                        if (bdist > 0 && bdist < 400) {
                            const repulse = (400 - bdist) * 0.0005;
                            this.vel.x -= (bx / bdist) * repulse;
                            this.vel.y -= (by / bdist) * repulse;
                        }
                    }

                    if (dist > 250) { 
                        const angle = Math.atan2(dy, dx);
                        this.vel.x += Math.cos(angle) * 0.05; 
                        this.vel.y += Math.sin(angle) * 0.05;
                    }
                    this.vel.x += dreadAvoidX;
                    this.vel.y += dreadAvoidY;
                    const speed = this.vel.mag();
                    let maxSpeed = this.type === 'heavy' ? 1.5 : (this.type === 'rapid' ? 3.0 : 2.5);
                    if (speed > maxSpeed) this.vel.mult(maxSpeed/speed);
                } else {
                    this.vel.mult(0.99);
                }

                this.pos.add(this.vel);
                checkDespawn(this, 6000); 
                this.shieldRotation += 0.01;
                this.innerShieldRotation -= 0.015;

                if (this.hp <= 5 && Math.random() < 0.1) spawnSmoke(this.pos.x, this.pos.y, 1);

                if (player && !player.dead) {
                    let px = player.pos.x, py = player.pos.y;
                    const dx = px - this.pos.x;
                    const dy = py - this.pos.y;
                    const dist = Math.hypot(dx, dy);
                    this.angle = Math.atan2(dy, dx);
                    this.turretAngle = Math.atan2(dy, dx);
                    this.angle += 0.002;

                    if (dist < 1500) {
                        this.shootTimer--;
                        if (this.shootTimer <= 0) {
                            const shootAngle = this.turretAngle;
                            if (this.type === 'heavy') {
                                for(let i=-1; i<=1; i++) {
                                    const a = shootAngle + i * 0.15;
                                    const bx = this.pos.x + Math.cos(a) * 75;
                                    const by = this.pos.y + Math.sin(a) * 75;
                                    bullets.push(new Bullet(bx, by, a, true, 3, 8, 8, '#fa0'));
                                    spawnBarrelSmoke(bx, by, a);
                                }
                                playSound('heavy_shoot');
                                this.shootTimer = 120;
                            } else if (this.type === 'rapid') {
                                const spread = (Math.random() - 0.5) * 0.1;
                                const a = shootAngle + spread;
                                const bx = this.pos.x + Math.cos(a) * 75;
                                const by = this.pos.y + Math.sin(a) * 75;
                                bullets.push(new Bullet(bx, by, a, true, 1, 14, 3, '#0ff'));
                                spawnBarrelSmoke(bx, by, a);
                                playSound('rapid_shoot');
                                this.shootTimer = 20; 
                            } else {
                                const damage = 2;
                                // 3-sided star shooting pattern
                                for(let i=0; i<3; i++) {
                                    const a = this.angle + i * (Math.PI * 2 / 3);
                                    const bx = this.pos.x + Math.cos(a) * 70;
                                    const by = this.pos.y + Math.sin(a) * 70;
                                    bullets.push(new Bullet(bx, by, a, true, damage));
                                    spawnBarrelSmoke(bx, by, a);
                                }
                                playSound('shoot');
                                this.shootTimer = difficultyTier >= 2 ? 40 : 60; 
                            }
                        }
                    }
                }
            }
            
            draw(ctx) {
                if(this.dead) return;
                ctx.save();
                ctx.translate(this.pos.x, this.pos.y);
                if (this.hp <= 2) ctx.translate((Math.random()-0.5)*2, (Math.random()-0.5)*2);
                
                let shieldColor = '#0ff';
                if (this.type === 'heavy') shieldColor = '#fa0';
                if (this.type === 'rapid') shieldColor = '#08f';

                ctx.save();
                ctx.rotate(this.shieldRotation);
                ctx.lineWidth = 4;
                const segCount = this.shieldSegments.length; 
                const segAngle = (Math.PI * 2) / segCount;
                for(let i=0; i<segCount; i++) {
                    if (this.shieldSegments[i] > 0) {
                         ctx.strokeStyle = shieldColor;
                         ctx.globalAlpha = Math.min(1.0, this.shieldSegments[i] / (this.maxShieldHp * 0.5)); 
                         ctx.shadowBlur = 10;
                         ctx.shadowColor = shieldColor;
                         ctx.beginPath();
                         ctx.arc(0, 0, this.shieldRadius, i * segAngle + 0.05, (i+1) * segAngle - 0.05);
                         ctx.stroke();
                         ctx.globalAlpha = 1.0;
                         ctx.shadowBlur = 0;
                         ctx.globalAlpha = 1.0;
                    }
                }
                ctx.restore();

                if (this.innerShieldSegments.length > 0) {
                    ctx.save();
                    ctx.rotate(this.innerShieldRotation);
                    ctx.lineWidth = 3;
                    const innerCount = this.innerShieldSegments.length;
                    const innerAngle = (Math.PI * 2) / innerCount;
                    let innerColor = '#f0f';
                    if (this.type === 'heavy') innerColor = '#f44';
                    if (this.type === 'rapid') innerColor = '#88f';
                    for(let i=0; i<innerCount; i++) {
                        if (this.innerShieldSegments[i] > 0) {
                             ctx.strokeStyle = innerColor;
                             ctx.globalAlpha = Math.min(1.0, this.innerShieldSegments[i] / (this.maxShieldHp * 0.5));
                             ctx.shadowBlur = 8;
                             ctx.shadowColor = innerColor;
                             ctx.beginPath();
                             ctx.arc(0, 0, this.innerShieldRadius, i * innerAngle + 0.05, (i+1) * innerAngle - 0.05);
                             ctx.stroke();
                             ctx.shadowBlur = 0;
                             ctx.globalAlpha = 1.0;
                        }
                    }
                    ctx.restore();
                }

                ctx.lineWidth = 3;
                if (this.isCruiser) {
                    // Battlestar-ish cruiser hull (rotated 180Â° via visualAngleOffset)
                    ctx.save();
                    const hs = this.cruiserHullScale || this.gunboatScale || 6;
                    const ang = this.angle + (this.visualAngleOffset || 0);
                    ctx.rotate(ang);
                    const rr = (x, y, w, h, r) => {
                        if (typeof ctx.roundRect === 'function') ctx.roundRect(x, y, w, h, r);
                        else ctx.rect(x, y, w, h);
                    };
                    
                    const moving = (this.vel && this.vel.mag && this.vel.mag() > 0.18);
                    const bodyLen = 40 * hs;
                    const bodyWid = 12 * hs;
                    
                    // Main hull (long slab with ridges)
                    ctx.shadowBlur = 16;
                    ctx.shadowColor = '#66ffff';
                    ctx.fillStyle = '#1a2230';
                    ctx.strokeStyle = '#66ffff';
                    ctx.lineWidth = 3;
                    ctx.beginPath();
                    rr(-bodyLen * 0.55, -bodyWid * 0.55, bodyLen * 1.1, bodyWid * 1.1, 10 * hs);
                    ctx.fill();
                    ctx.stroke();
                    
                    ctx.shadowBlur = 0;
                    ctx.globalAlpha = 0.65;
                    ctx.strokeStyle = '#88aaff';
                    for (let i = -2; i <= 2; i++) {
                        ctx.beginPath();
                        ctx.moveTo(-bodyLen * 0.45, i * bodyWid * 0.18);
                        ctx.lineTo(bodyLen * 0.55, i * bodyWid * 0.18);
                        ctx.stroke();
                    }
                    ctx.globalAlpha = 1;
                    
                    // Nose plate
                    ctx.fillStyle = '#0d131d';
                    ctx.strokeStyle = '#aaf';
